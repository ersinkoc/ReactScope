Quick Start

App.tsx 
import { ReactScopeProvider, Scope, useScope } from '@oxog/reactscope'

function App() {
  return (
    <ReactScopeProvider onReady={(kernel) => console.log('Ready!', kernel)}>
      <MyComponent />
    </ReactScopeProvider>
  )
}

function MyComponent() {
  const scope = useScope('MyComponent')

  return (
    <div>
      <p>Renders: {scope.renderCount}</p>
      <p>Avg Time: {scope.averageRenderTime.toFixed(2)}ms</p>
    </div>
  )
}
