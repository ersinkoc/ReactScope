Basic Usage

The simplest way to get started with ReactScope.

import React from 'react'
import { ReactScopeProvider, useScope } from '@oxog/reactscope'

function App() {
  return (
    <ReactScopeProvider>
      <Counter />
    </ReactScopeProvider>
  )
}

function Counter() {
  const [count, setCount] = React.useState(0)
  const scope = useScope('Counter')

  return (
    <div>
      <p>Count: {count}</p>
      <button onClick={() => setCount(c => c + 1)}>Increment</button>
      <p className="text-sm text-gray-500">
        Renders: {scope.renderCount} | Avg: {scope.averageRenderTime.toFixed(2)}ms
      </p>
    </div>
  )
}

Using Scope Component

Wrap any component without modifying its code.

import React from 'react'
import { Scope } from '@oxog/reactscope'

function App() {
  const [items, setItems] = React.useState([])

  return (
    <div>
      <Scope name="ExpensiveList" onRender={(m) => console.log(m)}>
        <ExpensiveList items={items} />
      </Scope>

      <Scope name="Sidebar">
        <Sidebar />
      </Scope>
    </div>
  )
}

Using Higher-Order Component

Add profiling to components using the withScope HOC.

import React from 'react'
import { withScope } from '@oxog/reactscope'

// Original component
function UserCard({ user }) {
  return (
    <div className="card">
      <img src={user.avatar} alt={user.name} />
      <h3>{user.name}</h3>
      <p>{user.email}</p>
    </div>
  )
}

// Enhanced with profiling
export const ProfiledUserCard = withScope(UserCard, {
  name: 'UserCard',
  trackProps: true,
  onRender: (metrics) => {
    if (metrics.wastedRenderCount > 0) {
      console.warn('UserCard has wasted renders!')
    }
  }
})

// Usage
function UserList({ users }) {
  return (
    <div>
      {users.map(user => (
        <ProfiledUserCard key={user.id} user={user} />
      ))}
    </div>
  )
}

Creating a Custom Plugin

Extend ReactScope with your own plugin.

import { createPlugin, ReactScopeProvider } from '@oxog/reactscope'

// Create a custom analytics plugin
const analyticsPlugin = createPlugin({
  name: 'analytics',
  version: '1.0.0',

  onInit: (kernel) => {
    console.log('Analytics plugin initialized')
  },

  hooks: {
    onRender: (event) => {
      // Send to analytics service
      if (event.duration > 16) {
        sendToAnalytics('slow_render', {
          component: event.componentName,
          duration: event.duration,
          timestamp: Date.now()
        })
      }
    },

    onMount: (event) => {
      sendToAnalytics('component_mount', {
        component: event.componentName,
        timestamp: Date.now()
      })
    }
  },

  api: {
    getSlowRenderCount: () => slowRenderCount,
    reset: () => { slowRenderCount = 0 }
  }
})

function App() {
  return (
    <ReactScopeProvider plugins={[analyticsPlugin]}>
      <YourApp />
    </ReactScopeProvider>
  )
}

Performance Monitoring Dashboard

Build a custom performance monitoring component.

import React from 'react'
import { useAllScopeMetrics, useScopeSummary } from '@oxog/reactscope'

function PerformanceMonitor() {
  const allMetrics = useAllScopeMetrics()
  const summary = useScopeSummary()

  const slowComponents = React.useMemo(() => {
    return Array.from(allMetrics.values())
      .filter(m => m.averageRenderTime > 16)
      .sort((a, b) => b.averageRenderTime - a.averageRenderTime)
      .slice(0, 10)
  }, [allMetrics])

  const wastedComponents = React.useMemo(() => {
    return Array.from(allMetrics.values())
      .filter(m => m.wastedRenderCount > 0)
      .sort((a, b) => b.wastedRenderCount - a.wastedRenderCount)
      .slice(0, 10)
  }, [allMetrics])

  return (
    <div className="p-4 bg-gray-900 rounded-lg">
      <h2 className="text-xl font-bold mb-4">Performance Overview</h2>

      <div className="grid grid-cols-4 gap-4 mb-6">
        <StatCard label="Components" value={summary.totalComponents} />
        <StatCard label="Total Renders" value={summary.totalRenders} />
        <StatCard
          label="Wasted Renders"
          value={summary.totalWastedRenders}
          warning={summary.totalWastedRenders > 10}
        />
        <StatCard
          label="Avg Render Time"
          value={`${summary.averageRenderTime.toFixed(2)}ms`}
          warning={summary.averageRenderTime > 16}
        />
      </div>

      <div className="grid grid-cols-2 gap-4">
        <div>
          <h3 className="font-semibold mb-2">Slowest Components</h3>
          <ul>
            {slowComponents.map(m => (
              <li key={m.componentId} className="text-sm">
                {m.componentName}: {m.averageRenderTime.toFixed(2)}ms
              </li>
            ))}
          </ul>
        </div>

        <div>
          <h3 className="font-semibold mb-2">Most Wasted Renders</h3>
          <ul>
            {wastedComponents.map(m => (
              <li key={m.componentId} className="text-sm">
                {m.componentName}: {m.wastedRenderCount} wasted
              </li>
            ))}
          </ul>
        </div>
      </div>
    </div>
  )
}

function StatCard({ label, value, warning }) {
  return (
    <div className={`p-3 rounded ${warning ? 'bg-red-900/50' : 'bg-gray-800'}`}>
      <div className="text-sm text-gray-400">{label}</div>
      <div className="text-2xl font-bold">{value}</div>
    </div>
  )
}

CI/CD Integration

Export metrics for performance regression testing.

import {
  ReactScopeProvider,
  createJsonExporterPlugin,
  getKernel
} from '@oxog/reactscope'

const jsonExporter = createJsonExporterPlugin({
  includeTimeline: true,
  pretty: true
})

function App() {
  return (
    <ReactScopeProvider
      plugins={[jsonExporter]}
      onReady={(kernel) => {
        // Export metrics after test run
        window.__exportMetrics = () => {
          const exporter = kernel.getPlugin('json-exporter')
          return exporter?.api?.export()
        }
      }}
    >
      <YourApp />
    </ReactScopeProvider>
  )
}

Conditional Profiling

Enable profiling only in development or for specific users.

import React from 'react'
import { ReactScopeProvider, createDashboardUIPlugin } from '@oxog/reactscope'

const isDev = process.env.NODE_ENV === 'development'
const isDebugUser = localStorage.getItem('debug') === 'true'

const plugins = isDev ? [createDashboardUIPlugin()] : []

function App() {
  return (
    <ReactScopeProvider
      enabled={isDev || isDebugUser}
      plugins={plugins}
      options={{
        sampleRate: isDev ? 1 : 0.1, // Sample 10% in production
        maxHistorySize: isDev ? 1000 : 100
      }}
    >
      <YourApp />
    </ReactScopeProvider>
  )
}
