API Reference

ReactScopeProvider

The root provider component that initializes the kernel and provides context to the component tree.

interface ReactScopeProviderProps {
  children: React.ReactNode
  enabled?: boolean              // Default: true
  plugins?: Plugin[]             // Additional plugins
  onReady?: (kernel: Kernel) => void
  options?: {
    trackAllComponents?: boolean // Default: false
    sampleRate?: number          // Default: 1 (0-1)
    maxHistorySize?: number      // Default: 1000
  }
}

Hooks

useScope

Main profiling hook for tracking component renders.

function useScope(name: string, options?: UseScopeOptions): ScopeHandle

interface ScopeHandle {
  readonly componentId: string
  readonly renderCount: number
  readonly lastRenderTime: number      // ms
  readonly averageRenderTime: number   // ms
  readonly wastedRenders: number
  readonly isTracking: boolean
  trackRender(duration: number, phase: 'mount' | 'update'): void
  trackPropsChange(changes: PropsDiff[]): void
}

useScopeMetrics

Read-only hook for accessing component metrics.

function useScopeMetrics(componentId: string): ComponentMetrics | undefined

useScopeContext

Access the kernel instance directly.

function useScopeContext(): Kernel

useAllScopeMetrics

Get metrics for all tracked components.

function useAllScopeMetrics(): Map<string, ComponentMetrics>

useScopeSummary

Get summary statistics across all components.

function useScopeSummary(): {
  totalComponents: number
  totalRenders: number
  totalWastedRenders: number
  averageRenderTime: number
}

Components

Scope

Wrapper component that profiles its children using React Profiler API.

interface ScopeProps {
  children: React.ReactNode
  name?: string
  onRender?: (metrics: ComponentMetrics) => void
  trackProps?: boolean
  trackState?: boolean
}

Higher-Order Components

withScope

HOC that adds profiling to any component.

function withScope<P extends object>(
  Component: React.ComponentType<P>,
  options?: WithScopeOptions
): React.ComponentType<P>

interface WithScopeOptions {
  name?: string
  trackProps?: boolean
  trackState?: boolean
  onRender?: (metrics: ComponentMetrics) => void
}

Kernel

interface Kernel {
  // Plugin management
  register(plugin: Plugin): void
  unregister(pluginName: string): void
  getPlugin<T extends Plugin>(name: string): T | undefined
  getPlugins(): Plugin[]

  // Event system
  emit(event: KernelEvent): void
  on<T extends KernelEvent['type']>(
    eventType: T,
    handler: (event: Extract<KernelEvent, { type: T }>) => void
  ): () => void

  // Metrics
  getMetrics(): MetricsStore
  updateMetrics(componentId: string, partial: Partial<ComponentMetrics>): void
  getComponentMetrics(componentId: string): ComponentMetrics | undefined
  clearMetrics(): void

  // Configuration
  configure(options: KernelOptions): void
  getOptions(): KernelOptions
  isEnabled(): boolean
  enable(): void
  disable(): void

  // Lifecycle
  destroy(): void
}
