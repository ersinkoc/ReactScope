ReactScopeProvider

The main provider component that initializes the ReactScope kernel and makes it available throughout your application.

Usage

import { ReactScopeProvider } from '@oxog/reactscope'

function App() {
  return (
    <ReactScopeProvider
      enabled={true}
      plugins={[]}
      onReady={(kernel) => console.log('Ready!')}
      options={{
        sampleRate: 1,
        maxHistorySize: 1000,
      }}
    >
      <YourApp />
    </ReactScopeProvider>
  )
}

Props

Prop Type Default Description
enabled boolean true Enable or disable profiling
plugins Plugin[] [] Additional plugins to register
onReady (kernel) => void undefined Callback when kernel is ready
options KernelOptions {} Kernel configuration options
children ReactNode required Your application components

Options

interface KernelOptions {
  // Sampling rate for events (0-1)
  sampleRate?: number

  // Maximum history size for metrics
  maxHistorySize?: number

  // Auto-track all components
  trackAllComponents?: boolean

  // Slow render threshold in ms
  slowRenderThreshold?: number
}

Example with Dashboard

import {
  ReactScopeProvider,
  createDashboardUIPlugin,
  createConsoleReporterPlugin
} from '@oxog/reactscope'

function App() {
  return (
    <ReactScopeProvider
      plugins={[
        createDashboardUIPlugin(),
        createConsoleReporterPlugin({ logLevel: 'warn' })
      ]}
      onReady={(kernel) => {
        // Show dashboard on startup
        kernel.getPlugin('dashboard-ui')?.api.show()
      }}
    >
      <YourApp />
    </ReactScopeProvider>
  )
}
