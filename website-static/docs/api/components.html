Components

ReactScope provides components for declarative performance tracking.

Scope

A wrapper component that tracks the performance of its children using React's Profiler API.

Usage

import { Scope } from '@oxog/reactscope'

function App() {
  return (
    <Scope
      name="ExpensiveList"
      trackProps={true}
      onRender={(metrics) => console.log(metrics)}
    >
      <ExpensiveList items={items} />
    </Scope>
  )
}

Props

Prop Type Default Description
name string required Unique identifier for this scope
trackProps boolean false Track props changes between renders
onRender (metrics) => void undefined Callback after each render
children ReactNode required Components to track

withScope (HOC)

A Higher-Order Component for wrapping class components or when you prefer the HOC pattern.

Usage

import { withScope } from '@oxog/reactscope'

function MyComponent({ data }) {
  return <div>{data}</div>
}

export default withScope(MyComponent, {
  name: 'MyComponent',
  trackProps: true,
  onRender: (metrics) => {
    console.log('Rendered in', metrics.duration, 'ms')
  }
})

Options

interface WithScopeOptions {
  // Component name for tracking
  name?: string

  // Track props changes
  trackProps?: boolean

  // Render callback
  onRender?: (metrics: RenderMetrics) => void
}

RenderMetrics Type

The metrics object passed to onRender callbacks: 

interface RenderMetrics {
  // Component identifier
  componentId: string
  componentName: string

  // Timing
  duration: number
  phase: 'mount' | 'update' | 'nested-update'
  timestamp: number

  // Counts
  renderCount: number
  wastedRenders: number

  // Averages
  averageRenderTime: number

  // Props diff (if trackProps enabled)
  propsChanged?: string[]
}
