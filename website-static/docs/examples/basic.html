Basic Usage Example

A simple example showing how to set up ReactScope and track component renders.

1. Setup Provider

// main.tsx
import React from 'react'
import ReactDOM from 'react-dom/client'
import { ReactScopeProvider } from '@oxog/reactscope'
import App from './App'

ReactDOM.createRoot(document.getElementById('root')!).render(
  <React.StrictMode>
    <ReactScopeProvider onReady={(k) => console.log('ReactScope ready', k)}>
      <App />
    </ReactScopeProvider>
  </React.StrictMode>
)

2. Track Components

// App.tsx
import { useScope } from '@oxog/reactscope'
import { useState } from 'react'

function App() {
  const scope = useScope('App')
  const [count, setCount] = useState(0)

  return (
    <div>
      <h1>ReactScope Basic Example</h1>

      <div className="metrics">
        <p>Render count: {scope.renderCount}</p>
        <p>Avg render time: {scope.averageRenderTime.toFixed(2)}ms</p>
        <p>Wasted renders: {scope.wastedRenders}</p>
      </div>

      <button onClick={() => setCount(c => c + 1)}>
        Count: {count}
      </button>

      <Counter />
    </div>
  )
}

function Counter() {
  const scope = useScope('Counter')
  const [value, setValue] = useState(0)

  return (
    <div>
      <h2>Counter Component</h2>
      <p>Value: {value}</p>
      <button onClick={() => setValue(v => v + 1)}>Increment</button>
      <small>Renders: {scope.renderCount}</small>
    </div>
  )
}

export default App

3. What You'll See

  • ✓ Each component shows its render count
  • ✓ Average render time is calculated automatically
  • ✓ Wasted renders are detected when props/state don't change
  • ✓ Counter updates don't affect App's render count
