Quick Start

import { ReactScopeProvider, useScope } from '@oxog/reactscope'

function App() {
  return (
    <ReactScopeProvider>
      <Counter />
    </ReactScopeProvider>
  )
}

function Counter() {
  const [count, setCount] = useState(0)
  const scope = useScope('Counter')

  return (
    <div>
      <p>Count: {count}</p>
      <button onClick={() => setCount(c => c + 1)}>+</button>
      <p>Renders: {scope.renderCount} | Avg: {scope.averageRenderTime.toFixed(2)}ms</p>
    </div>
  )
}
